# ====================================
# WIRED記事TOP5投稿Bot 環境変数設定
# ====================================
# 使用方法: このファイルの内容を .env に追加してください

# Gemini API設定
# 取得先: https://makersuite.google.com/app/apikey
GEMINI_API_KEY=your_gemini_api_key_here

# Bluesky設定
# ハンドル: あなたのBlueskyハンドル（例: username.bsky.social）
# パスワード: Blueskyアプリパスワード（設定 → アプリパスワード で作成）
BLUESKY_HANDLE=your_handle.bsky.social
BLUESKY_PASSWORD=your_app_password_here

# 投稿モード (bluesky または demo)
# demo: コンソールに表示のみ（投稿しない）
# bluesky: 実際にBlueskyに投稿
POST_MODE=demo

# ボット設定
# true: 改良版（記事本文を取得して詳細な要約）
# false: 基本版（RSSのsummaryのみ使用）
USE_ADVANCED_BOT=true

# テストモード
# true: 起動時に1回だけ実行してすぐ終了
# false: スケジュール実行（毎朝8時）
TEST_MODE=false
